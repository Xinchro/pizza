export const meta = { title: 'Home' };

export const navigateToStory = (storyId) => (e) => {
	e.preventDefault();
	window.parent.history.pushState({}, '', `/?path=/story/${storyId}`);
	window.parent.dispatchEvent(new PopStateEvent('popstate'));
};
